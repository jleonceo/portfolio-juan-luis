# ANÁLISIS DE VENTAS MENSUALES
# ============================================================
# Filtramos solo las líneas de ventas (cuenta 700)
# y vemos cómo evolucionan mes a mes durante 2024.
# Esto es lo que cualquier director comercial quiere ver
# de un vistazo — ¿cuándo vendemos más y menos?

# Filtramos ventas — cuenta 700, columna haber
ventas = df[df['cuenta'] == '700'].copy()

# Agrupamos por mes y sumamos el haber (importe de venta)
ventas_mes = ventas.groupby('mes')['haber'].sum().reset_index()
ventas_mes.columns = ['mes', 'total_ventas']

# Añadimos nombre del mes para que el gráfico sea más legible
meses = {1:'Ene', 2:'Feb', 3:'Mar', 4:'Abr', 5:'May', 6:'Jun',
         7:'Jul', 8:'Ago', 9:'Sep', 10:'Oct', 11:'Nov', 12:'Dic'}
ventas_mes['mes_nombre'] = ventas_mes['mes'].map(meses)

# Gráfico de barras — evolución mensual de ventas
fig, ax = plt.subplots(figsize=(12, 5))
bars = ax.bar(ventas_mes['mes_nombre'], ventas_mes['total_ventas'],
              color=AZUL, alpha=0.85, edgecolor='#1d4ed8')

# Añadimos el valor encima de cada barra
for bar, val in zip(bars, ventas_mes['total_ventas']):
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 1000,
            f'{val:,.0f}€', ha='center', va='bottom',
            fontsize=8, color='#cccccc')

ax.set_title('Ventas Mensuales 2024 — TechAcces SL', 
             fontsize=14, pad=15, color='white')
ax.set_xlabel('Mes', fontsize=11)
ax.set_ylabel('Importe (€)', fontsize=11)
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f'{x:,.0f}€'))
plt.tight_layout()
plt.show()

# Resumen numérico

print(f"📉 Mes con menos ventas: {ventas_mes.loc[ventas_mes['total_ventas'].idxmin(), 'mes_nombre']} ({ventas_mes['total_ventas'].min():,.2f}€)")

# ============================================================
# GRÁFICO EN CASCADA — CUENTA DE RESULTADOS
# ============================================================
# El gráfico en cascada (waterfall) es el estándar en finanzas
# para visualizar cómo se va construyendo el resultado.
# Cada barra muestra cómo una partida suma o resta al resultado.

categorias = ['Ventas', 'Compras', 'Nóminas', 'Gastos Op.', 
              'Amortiz.', 'Financiero', 'Resultado']
valores = [ventas_total, -compras_total, -nominas_total, 
           -gastos_total, -amortiz_total, -financiero_total, 
           resultado_final]

# Calculamos las bases para el efecto cascada
bases = [0]
acumulado = ventas_total
for v in valores[1:-1]:
    bases.append(acumulado if v < 0 else acumulado + v)
    acumulado += v
bases.append(0)  # El resultado final arranca desde 0

colores = [VERDE if v >= 0 else ROJO for v in valores]
colores[0] = AZUL   # Ventas en azul
colores[-1] = AMBAR  # Resultado en ámbar

fig, ax = plt.subplots(figsize=(12, 6))
bars = ax.bar(categorias, [abs(v) for v in valores], 
              bottom=bases, color=colores, alpha=0.85, edgecolor='#333')

# Valor encima/debajo de cada barra
for bar, val, base in zip(bars, valores, bases):
    y = base + abs(val) + 8000
    ax.text(bar.get_x() + bar.get_width()/2, y,
            f'{val:,.0f}€', ha='center', fontsize=8, color='#cccccc')

ax.set_title('Cuenta de Resultados en Cascada — TechAcces SL 2024',
             fontsize=14, pad=15, color='white')
ax.set_ylabel('Importe (€)', fontsize=11)
ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f'{x:,.0f}€'))

🔴 Problema Identificado — Déficit Estructural de Tesorería¶

El análisis de tesorería revela un problema crítico: TechAcces opera con flujo de caja negativo en 11 de los 12 meses del año.

¿Por qué ocurre esto?¶

Causa Detalle
Estacionalidad extrema El 44,7% de las ventas ocurren en Q4, pero los gastos son constantes los 12 meses

Solución 2 — Acelerar cobros B2B Reducir el plazo de pago de distribuidores de 60 a 30 días con descuentos por pronto pago. Impacto: mejora del flujo en Q1.

Solución 3 — Diversificar ventas fuera de Q4 Crear campañas en Q1 y Q2 para reducir la dependencia de Black Friday y Navidad. Objetivo: que Q1 pase del 15,7% al 20% de ventas.

# - Escenario Base: situación actual 2024
# - Escenario 1: refinanciación deuda -30%
# - Escenario 2: acelerar cobros B2B + refinanciación
# - Escenario 3: todo lo anterior + crecer ventas Q1/Q2 +20%

# ── PARÁMETROS BASE (situación actual) ──────────────────────
ventas_base       = 753263.05
compras_base      = 62575.05
nominas_base      = 325053.13
gastos_op_base    = 154513.26
amortiz_base      = 6280.08
financiero_base   = 262583.36

def cuenta_resultados(ventas, compras, nominas, gastos_op, amortiz, financiero, nombre):
    """
    Calcula la cuenta de resultados completa dados los parámetros.
    Devuelve un diccionario con todas las métricas clave.
    """
    margen_bruto   = ventas - compras
    ebitda         = margen_bruto - nominas - gastos_op
    ebit           = ebitda - amortiz
    resultado      = ebit - financiero

    return {
        'Escenario'     : nombre,
        'Ventas'        : ventas,
        'Compras'       : compras,
        'Margen Bruto'  : margen_bruto,
        '% Margen'      : margen_bruto / ventas * 100,
        'EBITDA'        : ebitda,
        '% EBITDA'      : ebitda / ventas * 100,
        'EBIT'          : ebit,
        'Financiero'    : financiero,
        'Resultado'     : resultado,
        '% Resultado'   : resultado / ventas * 100,
    }

# ── ESCENARIO BASE ───────────────────────────────────────────
base = cuenta_resultados(
    ventas_base, compras_base, nominas_base,
    gastos_op_base, amortiz_base, financiero_base,
    'Base 2024'
)

# ── ESCENARIO 1: Refinanciación deuda -30% ───────────────────
# Renegociamos el préstamo ICO — cuota mensual reducida un 30%
financiero_s1 = financiero_base * 0.70  # -30% gastos financieros
s1 = cuenta_resultados(
    ventas_base, compras_base, nominas_base,
    gastos_op_base, amortiz_base, financiero_s1,
    'S1: Refinanciación -30%'
)

# ── ESCENARIO 2: Refinanciación + cobros B2B más rápidos ─────
# Además reducimos compras un 10% por mejor negociación
compras_s2    = compras_base * 0.90
gastos_op_s2  = gastos_op_base * 0.95  # Pequeña mejora operativa
s2 = cuenta_resultados(
    ventas_base, compras_s2, nominas_base,
    gastos_op_s2, amortiz_base, financiero_s1,
    'S2: + Optimización costes'
)

# ── ESCENARIO 3: Todo lo anterior + crecimiento ventas ───────
# Crecemos ventas un 15% con campañas en Q1/Q2
ventas_s3 = ventas_base * 1.15
s3 = cuenta_resultados(
    ventas_s3, compras_s2, nominas_base,

)
